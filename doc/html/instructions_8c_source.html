127int ADD_8xy4(struct t_processor* processor, uint8_t x, uint8_t y)
128{
129 if (x >= 16 || y >= 16)
130 return 1;
131
132 // Créé une variable temporaire stockant Vx pour pouvoir l'étudier après l'opération
133 uint8_t temp = processor->generalRegister[x];
134 // Additionne Vx et Vy et stocke le résultat dans Vx
135 processor->generalRegister[x] += processor->generalRegister[y];
