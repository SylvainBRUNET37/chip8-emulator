96void loadRepresentativeSprite(struct t_RAM* RAM)
97{
98 // Déclaration d'un tableau contenant les sprites représentatifs des chiffres hexadécimaux
99 uint8_t representativeSprite[NB_REPRESENTATIVE_SPRITE] =
100 {
101 0xF0, 0x90, 0x90, 0x90, 0xF0, // 0

104 0xF0, 0x10, 0xF0, 0x10, 0xF0, // 3

106 0xF0, 0x80, 0xF0, 0x10, 0xF0, // 5
107 0xF0, 0x80, 0xF0, 0x90, 0xF0, // 6
108 0xF0, 0x10, 0x20, 0x40, 0x40, // 7
109 0xF0, 0x90, 0xF0, 0x90, 0xF0, // 8
110 0xF0, 0x90, 0xF0, 0x10, 0xF0, // 9
111 0xF0, 0x90, 0xF0, 0x90, 0x90, // A
112 0xE0, 0x90, 0xE0, 0x90, 0xE0, // B
113 0xF0, 0x80, 0x80, 0x80, 0xF0, // C
114 0xE0, 0x90, 0x90, 0x90, 0xE0, // D
115 0xF0, 0x80, 0xF0, 0x80, 0xF0, // E
116 0xF0, 0x80, 0xF0, 0x80, 0x80 // F
117 };
118
119 // Écrit dans la RAM les réprésentation hexa des sprites dans les NB_REPRESENTATIVE_SPRITE premiers emplacements
120 for (uint8_t i = 0 ; i < NB_REPRESENTATIVE_SPRITE ; i++)
121 writeRAM(RAM, i, representativeSprite[i]);
122}

128 uint16_t instruction = readRAM(machine->RAM, machine->processor->programCounter);
129 instruction = instruction << 8;
130 instruction += readRAM(machine->RAM, machine->processor->programCounter+1);
131 machine->processor->programCounter += 2;

133 // DECODE/EXECUTE :
134 // - les conditions dans les "if / else if" vérifient quel instruction a été récupérée en RAM
